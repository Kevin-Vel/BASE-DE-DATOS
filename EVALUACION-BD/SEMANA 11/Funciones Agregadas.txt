Comando SUM: Te da el total de una columna de tu tabla
SELECT SUM(columna) AS Total
FROM tabla;

Comando Count: Cuenta el numero de registros en la tabla
SELECT COUNT(*) AS Cantidad
FROM tabla;
y Para valores nulos hacemos lo siguiente:
SELECT COUNT(columna) AS CantidadNoNula
FROM tabla;

Comando AVG: Calcular el promedio de una colummna de tu tabla
SELECT AVG(columna) AS Promedio
FROM tabla;

Comando Max: obtiene el valor maximo de una de tus columnas
SELECT MAX(columna) AS Maximo
FROM tabla;

Comando Min: Obtiene el valor minimo de una de tus colummnas
SELECT MIN(columna) AS Minimo
FROM tabla;
